Win32:

Launch in a VS2017 x64 x86 Cross Tools Command Prompt:
Make sure python 2.7 is in the path, not 3.x
Download ninja and copy in bin

For debug build: 

python tools/git-sync-deps

bin\gn.exe gen out/Debug --args="is_official_build=false is_component_build=true is_debug=true skia_use_freetype=true skia_use_fontconfig=true skia_enable_fontmgr_win=false skia_use_lua=false skia_enable_gpu=false skia_use_libheif=false skia_enable_pdf=false skia_use_system_freetype2=true clang_win=\"C:\Program Files\LLVM\" target_cpu=\"x86\" extra_ldflags=[\"/LIBPATH:C:\SpinetiX\sources\genesis\trunk\build\win32\Debug\lib\"] extra_cflags=[\"-IC:\SpinetiX\sources\genesis\trunk\build\win32\Debug\include\",\"-IC:\SpinetiX\sources\genesis\trunk\build\win32\Debug\include\freetype2\"]"
bin\ninja -C out/Debug

For release build:

bin\gn.exe gen out/Release --args="is_official_build=false is_component_build=true is_debug=false skia_use_freetype=true skia_use_fontconfig=true skia_enable_fontmgr_win=false skia_use_lua=false skia_enable_gpu=false skia_use_libheif=false skia_enable_pdf=false skia_use_system_freetype2=true clang_win=\"C:\Program Files\LLVM\" target_cpu=\"x86\" extra_ldflags=[\"/LIBPATH:C:\SpinetiX\sources\genesis\trunk\build\win32\Release\lib\"] extra_cflags=[\"-IC:\SpinetiX\sources\genesis\trunk\build\win32\Release\include\",\"-IC:\SpinetiX\sources\genesis\trunk\build\win32\Release\include\freetype2\"]"
bin\ninja -C out/Release


Linux-x64:

git clone 'https://chromium.googlesource.com/chromium/tools/depot_tools.git'
export PATH="${PWD}/depot_tools:${PATH}"
git clone https://github.com/spinetix/skia
cd skia
git checkout spx/m79
python tools/git-sync-deps
bin/gn gen out/Debug --args='is_official_build=false is_component_build=true is_debug=true skia_enable_gpu=false skia_use_icu=false cc="/home/SPINETIX/jcm/clang+llvm-9.0.0-x86_64-linux-gnu-ubuntu-14.04/bin/clang" cxx="/home/SPINETIX/jcm/clang+llvm-9.0.0-x86_64-linux-gnu-ubuntu-14.04/bin/clang++"'
ninja -C out/Debug
bin/gn gen out/Release --args='is_official_build=false is_component_build=true is_debug=false skia_enable_gpu=false skia_use_icu=false cc="/home/SPINETIX/jcm/clang+llvm-9.0.0-x86_64-linux-gnu-ubuntu-14.04/bin/clang" cxx="/home/SPINETIX/jcm/clang+llvm-9.0.0-x86_64-linux-gnu-ubuntu-14.04/bin/clang++"'
ninja -C out/Release
